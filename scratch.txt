schema.prisma

model Customer {
  id        Int      @id @default(autoincrement())
  name     String   
  address   String?
  phoneNumber String  

  Cart Cart[]
}

model Cart {
  id     Int     @id @default(autoincrement())
  cartNumber    Int
  userId Int     @unique
  productId Int     @unique

  Customer Customer @relation(fields: [userId], references: [id])
  Product Product @relation(fields: [productId], references: [id])
}

model Product {
  id      Int      @id @default(autoincrement())
  name    String?
  cost   Int
  count Int
  Cart Cart[]
}
